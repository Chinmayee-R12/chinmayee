#!/bin/bash
echo "Hello from github project"
echo "CI/CD pipeline executed successfully"





pipeline {
  agent any

  stages {
    stage('Clone') {
      steps {
        git url: 'https://github.com/Chinmayee-R12/jenkins-simple-demo.git',
          branch: 'main'
      }
    }
    stage('Run Script') {
      steps {
        sh 'chmod +x script.sh'
        sh './script.sh'
      }
    }
  }
}
